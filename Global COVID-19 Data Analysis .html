```python import pandas as pd import matplotlib.pyplot as plt import seaborn as sns # Load data url = 'https://covid.ourworldindata.org/data/owid-covid-data.csv' df = pd.read_csv(url) # Explore data print(df.head()) print(df.columns) # Filter to relevant columns df = df[['location', 'date', 'total_cases', 'new_cases', 'total_deaths', 'continent']] # Drop rows with missing continent df = df.dropna(subset=['continent']) # Convert date column to datetime df['date'] = pd.to_datetime(df['date']) # Group by continent and date continent_df = df.groupby(['continent', 'date'])[['new_cases', 'total_cases', 'total_deaths']].sum().reset_index() # Plotting total cases over time by continent plt.figure(figsize=(14, 6)) sns.lineplot(data=continent_df, x='date', y='total_cases', hue='continent') plt.title('Total COVID-19 Cases by Continent Over Time') plt.xlabel('Date') plt.ylabel('Total Cases') plt.xticks(rotation=45) plt.tight_layout() plt.show()
